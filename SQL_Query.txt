(SELECT image_id, 
	CASE 
           WHEN x.score >= 0.5 THEN 1
           ELSE 0
       END AS WEAK_LABEL FROM 
         (SELECT IMAGE_ID, SCORE, ROW_NUMBER() OVER (ORDER BY SCORE DESC) AS T FROM UNLABELED_IMAGE_PREDICTIONS) AS X WHERE X.T % 3 = 1 LIMIT 10000)
         UNION
         (SELECT image_id, 
	CASE 
           WHEN x.score >= 0.5 THEN 1
           ELSE 0
       END AS WEAK_LABEL FROM 
         (SELECT IMAGE_ID, SCORE, ROW_NUMBER() OVER (ORDER BY SCORE ASC) AS T FROM UNLABELED_IMAGE_PREDICTIONS) AS X WHERE X.T % 3 = 1 LIMIT 10000);
